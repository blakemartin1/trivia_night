    async function submitAnswer() {
        const team = document.getElementById('teamName').value.trim();
        const ans = document.getElementById('playerAnswer').value.trim();

        if (!team || !ans) {
            alert("Please fill out your team name and answer!");
            return;
        }

        const { error } = await supabaseClient
            .from('submissions')
            .insert([{ team_name: team, answer_submitted: ans, round_number: 1, question_number: 1 }]);

        if (!error) {
            document.getElementById('playerAnswer').value = ''; 
            const banner = document.getElementById('success');
            banner.style.display = 'block';
            setTimeout(() => banner.style.display = 'none', 3000);
        } else {
            alert("Error sending answer.");
            console.error(error);
        }
    }
